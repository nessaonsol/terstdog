‚öõÔ∏è Frontend Architecture

The Gorbot interface is built using modern web technologies with a focus on clean green design and smooth user experience:

React + TypeScript ChatInterface.tsx 
import React, { useState, useEffect } from 'react';
import { WebSocket } from 'ws';

interface Message {
  id: string;
  author: 'user' | 'Gorbot';
  content: string;
  timestamp: Date;
}

export const ChatInterface: React.FC = () => {
  const [messages, setMessages] = useState<Message[]>([]);
  const [socket, setSocket] = useState<WebSocket | null>(null);
  
  useEffect(() => {
    // Connect to ElizaOS backend via WebSocket
    const ws = new WebSocket('ws://localhost:3000/chat');
    
    ws.onmessage = (event) => {
      const response = JSON.parse(event.data);
      setMessages(prev => [...prev, {
        id: response.id,
        author: 'Gorbot',
        content: response.text,
        timestamp: new Date(response.timestamp)
      }]);
    };
    
    setSocket(ws);
    return () => ws.close();
  }, []);
  
  const sendMessage = (content: string) => {
    if (socket && socket.readyState === WebSocket.OPEN) {
      const message = {
        text: content,
        userId: generateUserId(),
        timestamp: new Date().toISOString()
      };
      
      socket.send(JSON.stringify(message));
      
      // Add user message to UI immediately
      setMessages(prev => [...prev, {
        id: generateId(),
        author: 'user', 
        content,
        timestamp: new Date()
      }]);
    }
  };
  
  return (
    <div className="chat-interface">
      {/* Clean green chat container */}
      <div className="chat-container">
        {messages.map(message => (
          <MessageBubble key={message.id} message={message} />
        ))}
      </div>
      <ChatInput onSend={sendMessage} />
    </div>
  );
};

üîå Extending Gorbot

Gorbot is designed to be extensible. Developers can add new plugins, personality traits, or integrations:

// Example: Adding a productivity plugin
export const productivityPlugin: Plugin = {
  name: "productivity",
  description: "Helps users with productivity and task management",
  
  actions: [
    {
      name: "CREATE_TASK", 
      similes: ["add task", "remind me", "create todo"],
      examples: [
        [
          {
            user: "User",
            content: { text: "Remind me to call John tomorrow" }
          },
          {
            user: "Gorbot",
            content: { 
              text: "Task created: Call John (Due: Tomorrow) ‚úÖ",
              action: "CREATE_TASK"
            }
          }
        ]
      ],
      
      handler: async (runtime, memory, state, message) => {
        const task = extractTaskFromMessage(message.content.text);
        const newTask = await createTask(task);
        
        return {
          text: `Task created: ${newTask.title} ${newTask.dueDate ? `(Due: ${formatDate(newTask.dueDate)})` : ''} ‚úÖ`,
          action: "CREATE_TASK"
        };
      }
    }
  ]
};

// Register the plugin
runtime.registerPlugin(productivityPlugin);
